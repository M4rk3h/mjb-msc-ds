# load tweets as a corpus
df <- Corpus(VectorSource(df$text))

# Clean the text
getTransformations()
## [1] "removeNumbers"     "removePunctuation" "removeWords"      
## [4] "stemDocument"      "stripWhitespace"
myTweets <- tm_map(df, content_transformer(tolower))
## Warning in tm_map.SimpleCorpus(df, content_transformer(tolower)): transformation
## drops documents
myTweets <- tm_map(df, removeNumbers)
## Warning in tm_map.SimpleCorpus(df, removeNumbers): transformation drops
## documents
myTweets <- tm_map(df, removeWords, stopwords(kind = "en"))
## Warning in tm_map.SimpleCorpus(df, removeWords, stopwords(kind = "en")):
## transformation drops documents
myTweets <- tm_map(df, removePunctuation)
## Warning in tm_map.SimpleCorpus(df, removePunctuation): transformation drops
## documents
myTweets <- tm_map(df, stripWhitespace)
## Warning in tm_map.SimpleCorpus(df, stripWhitespace): transformation drops
## documents
myTweets <- tm_map(df, stemDocument)
## Warning in tm_map.SimpleCorpus(df, stemDocument): transformation drops documents
# custom stop words
myTweets <- tm_map(df, removeWords, c("and", "the", "how"))
